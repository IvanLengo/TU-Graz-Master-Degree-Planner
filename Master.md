  md += ` <font color="#ffc000">Realised ECTS: ${realisedECTS}</font><br><font color="#c00000">Set ECTS: ${setECTS}</font> |`;
}
md += '\n';



md += '|';
for (let sem of semesters) {
  const list = courses.filter(c => Number(c.sem) === sem);

  if (list.length > 0) {
    let sum = 0;
    let count = 0;

    for (const c of list) {
      let val = 0;
      if (typeof c.difficulty === 'number' && c.difficulty > 0) {
        val = c.difficulty;
      } else if (typeof c.difficulty === 'string') {
        // stars as string "⭐⭐⭐"
        const stars = (c.difficulty.match(/⭐/g) || []).length;
        if (stars > 0) val = stars;
        // numeric string "3"
        else if (!isNaN(parseFloat(c.difficulty))) val = parseFloat(c.difficulty);
      }
      // fallback: maybe stored in stat as stars
      if (!val && c.stat && typeof c.stat === 'string') {
        const stars = (c.stat.match(/⭐/g) || []).length;
        if (stars > 0) val = stars;
      }

      if (val > 0) {
        sum += val;
        count++;
      }
    }

    const avg = count > 0 ? sum / count : 0;
    md += ` Avg. Difficulty: ${avg.toFixed(1)} / 5 ⭐ |`;
  } else {
    md += ' |';
  }
}
md += '\n';


return engine.markdown.create(md);

  // find max rows needed
  const maxRows = Math.max(...semesters.map(s => group.filter(c => Number(c.sem) === s).length));
